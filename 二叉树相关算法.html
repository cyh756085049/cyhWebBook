144. 二叉树的前序遍历

给定一个二叉树，返回它的 前序 遍历。

输入: [1,null,2,3]  
   1
    \
     2
    /
   3 

输出: [1,2,3]

思路：递归、迭代（栈）

前序遍历是根左右，使用迭代方法，可以使用数组模拟栈，当根节点不为空时，将根节点入栈，先访问根节点，然后依次访问根节点的右节点和左节点，打印的时候是先是左节点，然后是右节点。

/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/** 1、递归法
 * @param {TreeNode} root
 * @return {number[]}
 */
var preorderTraversal = function(root) {
    const res = [];
    function traversal(root) {
        if (root !== null) {
            res.push(root.val);
            traversal(root.left);
            traversal(root.right);
        }
    }
    traversal(root);
    return res;
};

/** 2、迭代法
 * @param {TreeNode} root
 * @return {number[]}
 */
var preorderTraversal = function(root) {
    const res = [];
    // 数组模拟栈
    const stack = [];
    // 当根节点不为空时，将根节点入栈
    if (root) stack.push(root);
    while (stack.length > 0) {
        const node = stack.pop();
        // 先访问的是根节点
        res.push(node.val);
        // 栈中加入的是先右节点，后左节点，打印的时候是先左节点，后右节点
        if (node.right !== null) {
            stack.push(node.right);
        }
        if (node.left !== null) {
            stack.push(node.left);
        }
    }
    return res;
};

94. 二叉树的中序遍历

给定一个二叉树，返回它的中序 遍历。

输入: [1,null,2,3]
   1
    \
     2
    /
   3

输出: [1,3,2]

思路：递归、迭代（栈）

中序遍历是左根右，在迭代法中，首先遍历根节点的左节点，直到左节点为空时，出栈该左节点，然后访问其右节。

/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/** 1、递归法
 * @param {TreeNode} root
 * @return {number[]}
 */
var inorderTraversal = function(root) {
    const res = [];
    function traversal(root) {
        if (root !== null) {
            traversal(root.left);
            res.push(root.val);
            traversal(root.right);
        }
    }
    traversal(root);
    return res;
};

/** 2、迭代法
 * @param {TreeNode} root
 * @return {number[]}
 */
var inorderTraversal = function(root) {
    const res = [];
    const stack = [];
    while (stack.length || root) {
        while (root) {
            stack.push(root);
            root = root.left;
        }
        root = stack.pop();
        res.push(root.val);
        root = root.right; 
    }
    return res;
};

145. 二叉树的后序遍历

给定一个二叉树，返回它的 后序 遍历。

输入: [1,null,2,3]  
   1
    \
     2
    /
   3 

输出: [3,2,1]

思路：递归、迭代（栈）

后序遍历是左右根，迭代法可以结合先序遍历，先序遍历是根左右，则逆序之后是右左根，然后可以把右左根转化成根左右根即可，使用unshift()方法。

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */

/** 1、递归法
 * @param {TreeNode} root
 * @return {number[]}
 */
var postorderTraversal = function(root) {
    const res = [];
    function traversal(root) {
        if (root) {
            traversal(root.left);
            traversal(root.right);
            res.push(root.val);
        }
    }
    traversal(root);
    return res;
};

/** 2、迭代法
 * @param {TreeNode} root
 * @return {number[]}
 */
var postorderTraversal = function(root) {
    const res = [], stack = [];
    if (root) stack.push(root);
    while (stack.length) {
        const node = stack.pop();
        res.unshift(node.val);
        if (node.left) stack.push(node.left);
        if (node.right) stack.push(node.right);
    }
    return res;
};

102. 二叉树的层序遍历

给你一个二叉树，请你返回其按 层序遍历 得到的节点值。 （即逐层地，从左到右访问所有节点）。

二叉树：[3,9,20,null,null,15,7],
        3
   / \
  9  20
    /  \
   15   7
返回其层次遍历结果：
[
  [3],
  [9,20],
  [15,7]
]

思路：递归

/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/** 递归法
 * @param {TreeNode} root
 * @return {number[][]}
 */
 var invertTree = function(root) {
     const res = [];
    function traversal (root, depth) {
        if (root) {
            if (!res[depth]) {
                res[depth] = [];
            }
            traversal(root.left, depth + 1);
            res[depth].push(root.val);
            traversal(root.right, depth + 1);
        }
    }
    traversal(root, 0);
    return res;
}
